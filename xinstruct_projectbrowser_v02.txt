@app.route('/api/action_jobactive_dashboard_update', methods=['POST'])
def api_action_jobactive_dashboard_update():
    """Update job database entry with edited values"""
    try:
        data = request.json
        job_name = data.get('job_name')  # Primary key for UPDATE WHERE clause
        
        if not job_name:
            return jsonify({'success': False, 'message': 'Job name is required'}), 400
        
        # Connect to database
        conn = dbj.db_get_connection(path_db_jobs)
        cursor = conn.cursor()
        
        # Update the job record
        cursor.execute('''
            UPDATE projects SET
                job_alias = ?,
                job_state = ?,
                job_year = ?,
                job_user_id = ?,
                job_user_name = ?,
                job_date_created = ?,
                job_edit_user_id = ?,
                job_edit_user_name = ?,
                job_edit_date = ?,
                job_date_start = ?,
                job_date_due = ?,
                job_charge1 = ?,
                job_charge2 = ?,
                job_charge3 = ?,
                job_path_job = ?,
                job_path_rnd = ?,
                job_apps = ?,
                job_tags = ?,
                job_notes = ?
            WHERE job_name = ?
        ''', (
            data.get('job_alias', ''),
            data.get('job_state', ''),
            data.get('job_year', ''),
            data.get('job_user_id', ''),
            data.get('job_user_name', ''),
            data.get('job_date_created', ''),
            data.get('job_edit_user_id', ''),
            data.get('job_edit_user_name', ''),
            data.get('job_edit_date', ''),
            data.get('job_date_start', ''),
            data.get('job_date_due', ''),
            data.get('job_charge1', ''),
            data.get('job_charge2', ''),
            data.get('job_charge3', ''),
            data.get('job_path_job', ''),
            data.get('job_path_rnd', ''),
            data.get('job_apps', ''),
            data.get('job_tags', ''),
            data.get('job_notes', ''),
            job_name  # WHERE clause
        ))
        
        conn.commit()
        rows_affected = cursor.rowcount
        conn.close()
        
        if rows_affected == 0:
            return jsonify({'success': False, 'message': f'Job "{job_name}" not found in database'}), 404
        
        return jsonify({'success': True, 'message': f'Job "{job_name}" updated successfully'})
        
    except sqlite3.Error as e:
        return jsonify({'success': False, 'message': f'Database error: {str(e)}'}), 500
    except Exception as e:
        return jsonify({'success': False, 'message': f'Server error: {str(e)}'}), 500
